#### 分页插件

使用jQuery插件设置分页效果

官网地址: http://josecebe.github.io/twbs-pagination/

```javascript
$list.twbsPagination({
    totalPages : totalPages, // 总页数
    visiblePages : 7, // 可见页数
    onPageClick : function (event, page) {
      
    },
    first : '首页',  // 设置首页的文字
    last : '末页',
    next : '下一页',
    prev : '上一页'
});
```

模块化：

```javascript
//  主页面(需要模块化操作的页面)引入require.js
<script src="require.js" data-main = "main.js"> 
    
//  main.js 如下
// 使用config方法进行配置设置
//	 设置paths后,在[]中书写的就是模块的标识名称了
//			- 不符合规范的模块不能像其他模块一样进行引入,需要单独设置
require.config({
	baseUrl : '/static/assets/vendors/',
	paths : {
		'jquery' : 'jquery/jquery',
		'template' : 'art-template/template-web',
		'bootstrap' : 'bootstrap/js/bootstrap.min',
		'twbsPagination' : 'twbs-pagination/jquery.twbsPagination.min'
	},
  // shim时用来定义不符合规范模块的设置
	shim : {
		'twbsPagination' : {
			deps : ['jquery'], // 用于声明当前模块的依赖关系
			exports : '$.fn.twbsPagination' // 用于声明当前模块输出的功能
		}
	}
});
// 参数1： 所以需要依赖的模块(js文件)
// 参数2： 依赖函数所提供的功能。
require.(['jquery','template','twbsPagination'],function ($,template) {
       //  用来放置当前模块的主要功能
  });
```
